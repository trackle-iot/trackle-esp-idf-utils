36/* Event handler for catching system events */
37static void bt_event_handler(void *arg, esp_event_base_t event_base,
38 int32_t event_id, void *event_data)
39{
40 ESP_LOGI(BT_TAG, "----------------------------------------");
41 ESP_LOGI(BT_TAG, "bt event_handler: %s %d", event_base, event_id);
42 ESP_LOGI(BT_TAG, "----------------------------------------");
43
44 if (event_base == WIFI_PROV_EVENT)
45 {
46 switch (event_id)
47 {
48 case WIFI_PROV_START:
49 ESP_LOGI(BT_TAG, "Provisioning started");
50 xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
51 xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_RUN);
52 break;
53 case WIFI_PROV_CRED_RECV:
54 {
55 wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t *)event_data;
56 ESP_LOGI(BT_TAG, "Received Wi-Fi credentials"
57 "\n\tSSID : %s\n\tPassword : %s",
58 (const char *)wifi_sta_cfg->ssid,
59 (const char *)wifi_sta_cfg->password);
60 break;
61 }
62 case WIFI_PROV_CRED_FAIL:
63 {
64 wifi_prov_sta_fail_reason_t *reason = (wifi_prov_sta_fail_reason_t *)event_data;
65 ESP_LOGE(BT_TAG, "Provisioning failed!\n\tReason : %s"
66 "\n\tPlease reset to factory and retry provisioning",
67 (*reason == WIFI_PROV_STA_AUTH_ERROR) ? "Wi-Fi station authentication failed" : "Wi-Fi access-point not found");
68
69 prov_retry_num++;
70 if (prov_retry_num >= PROV_MGR_MAX_RETRY_CNT)
71 {
72 ESP_LOGI(BT_TAG, "Failed to connect with provisioned AP, reseting provisioned credentials and restarting...");
73 wifi_config_t wifi_cfg = {0};
74 esp_err_t err = esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg);
75 if (err != ESP_OK)
76 {
77 ESP_LOGE(BT_TAG, "Failed to set wifi config, 0x%x", err);
78 }
79 xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
80 xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_ERR);
81 xEventGroupSetBits(s_wifi_event_group, RESTART);
82 }
83
84 break;
85 }
86 case WIFI_PROV_CRED_SUCCESS:
87 ESP_LOGI(BT_TAG, "Provisioning successful");
88 xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
89 xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_OK);
90 break;
91 case WIFI_PROV_END:
92 // De-initialize manager once provisioning is finished and restart
93 ESP_LOGI(BT_TAG, "Provisioning end");
94 wifi_prov_mgr_deinit();
95 xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
96 xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_NO);
97 xEventGroupSetBits(s_wifi_event_group, RESTART);
98 break;
99 default:
100 break;
101 }
102 }
103
104 ESP_LOGI(BT_TAG, "end bt_event_handler: -------------------");
105}

121static int btPostCbClaimCode(const char *args)
122{
123 char *key = strtok(args, ",");
124 if (key == NULL || strcmp(key, "cc") != 0)
125 {
126 ESP_LOGE("cc", "Invalid key for setting claim code");
127 return -1;
128 }
129 char *claimCode = strtok(NULL, ",");
130 if (key == NULL || strlen(claimCode) != 63)
131 {
132 ESP_LOGE("cc", "Invalid claim code");
133 return -1;
134 }
135 ESP_LOGE("cc", "Claim code received successfully:");
136 ESP_LOG_BUFFER_CHAR_LEVEL("cc", claimCode, CLAIM_CODE_LENGTH, ESP_LOG_ERROR);
137 trackleSetClaimCode(trackle_s, claimCode);
138 Trackle_saveClaimCode(claimCode);
139 return 1;
140}

156void trackle_utils_bt_provision_loop()
157{
158 EventBits_t bits = xEventGroupGetBits(s_wifi_event_group);
159
160 // Check if start or stop provisioning
161 if (bits & START_PROVISIONING)
162 {
163 xEventGroupClearBits(s_wifi_event_group, START_PROVISIONING);
164 xEventGroupSetBits(s_wifi_event_group, IS_PROVISIONING);
165
166 esp_wifi_set_ps(WIFI_PS_MIN_MODEM); // enable powersave
167
168 // Configuration for the provisioning manager
169 wifi_prov_mgr_config_t config = {
170 .scheme = wifi_prov_scheme_ble,
171 .scheme_event_handler = WIFI_PROV_SCHEME_BLE_EVENT_HANDLER_FREE_BTDM,
172 };
173
174 // Initialize provisioning manager
175 wifi_prov_mgr_init(config);
176
177 btFunctionsEndpointsCreate();
178
179 char service_name[14];
180 get_device_service_name(service_name, sizeof(service_name));
181
182 wifi_prov_security_t security = WIFI_PROV_SECURITY_1;
183
184 const char *service_key = NULL;
185 uint8_t custom_service_uuid[] = {0xb4, 0xdf, 0x5a, 0x1c, 0x3f, 0x6b, 0xf4, 0xbf, 0xea, 0x4a, 0x82, 0x03, 0x04, 0x90, 0x1a, 0x02};
186 wifi_prov_scheme_ble_set_service_uuid(custom_service_uuid);
187
188 esp_err_t prov_err = wifi_prov_mgr_start_provisioning(security, NULL, service_name, service_key);
189 ESP_LOGI(BT_TAG, "wifi_prov_mgr_start_provisioning %d", prov_err);
190 btFunctionsEndpointsRegister();

17/* Event handler for catching system events */
18static void bt_event_handler(void *arg, esp_event_base_t event_base,
19 int32_t event_id, void *event_data)
20{
21 ESP_LOGI(BT_TAG, "----------------------------------------");
22 ESP_LOGI(BT_TAG, "bt event_handler: %s %d", event_base, event_id);
23 ESP_LOGI(BT_TAG, "----------------------------------------");
24
25 if (event_base == WIFI_PROV_EVENT)
26 {
27 switch (event_id)
28 {
29 case WIFI_PROV_START:
30 ESP_LOGI(BT_TAG, "Provisioning started");
31 break;
32 case WIFI_PROV_CRED_RECV:
33 {
34 wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t *)event_data;
35 ESP_LOGI(BT_TAG, "Received Wi-Fi credentials"
36 "\n\tSSID : %s\n\tPassword : %s",
37 (const char *)wifi_sta_cfg->ssid,
38 (const char *)wifi_sta_cfg->password);
39 break;
40 }
41 case WIFI_PROV_CRED_FAIL:
42 {
43 wifi_prov_sta_fail_reason_t *reason = (wifi_prov_sta_fail_reason_t *)event_data;
44 ESP_LOGE(BT_TAG, "Provisioning failed!\n\tReason : %s"
45 "\n\tPlease reset to factory and retry provisioning",
46 (*reason == WIFI_PROV_STA_AUTH_ERROR) ? "Wi-Fi station authentication failed" : "Wi-Fi access-point not found");
47
48 prov_retry_num++;
49 if (prov_retry_num >= PROV_MGR_MAX_RETRY_CNT)
50 {
51 ESP_LOGI(BT_TAG, "Failed to connect with provisioned AP, reseting provisioned credentials and restarting...");
52 wifi_config_t wifi_cfg = {0};
53 esp_err_t err = esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg);
54 if (err != ESP_OK)
55 {
56 ESP_LOGE(BT_TAG, "Failed to set wifi config, 0x%x", err);
57 }
58
59 xEventGroupSetBits(s_wifi_event_group, RESTART);
60 }
61
62 break;
63 }
64 case WIFI_PROV_CRED_SUCCESS:
65 ESP_LOGI(BT_TAG, "Provisioning successful");
66 break;
67 case WIFI_PROV_END:
68 // De-initialize manager once provisioning is finished and restart
69 ESP_LOGI(BT_TAG, "Provisioning end");
70 wifi_prov_mgr_deinit();
71 xEventGroupSetBits(s_wifi_event_group, RESTART);
72 break;
73 default:
74 break;
75 }
76 }
77
78 ESP_LOGI(BT_TAG, "end bt_event_handler: -------------------");
79}

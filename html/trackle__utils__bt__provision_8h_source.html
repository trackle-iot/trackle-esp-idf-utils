18/* Event handler for catching system events */
19static void bt_event_handler(void *arg, esp_event_base_t event_base,
20 int32_t event_id, void *event_data)
21{
22 ESP_LOGI(BT_TAG, "----------------------------------------");
23 ESP_LOGI(BT_TAG, "bt event_handler: %s %d", event_base, event_id);
24 ESP_LOGI(BT_TAG, "----------------------------------------");
25
26 if (event_base == WIFI_PROV_EVENT)
27 {
28 switch (event_id)
29 {
30 case WIFI_PROV_START:
31 ESP_LOGI(BT_TAG, "Provisioning started");
32 break;
33 case WIFI_PROV_CRED_RECV:
34 {
35 wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t *)event_data;
36 ESP_LOGI(BT_TAG, "Received Wi-Fi credentials"
37 "\n\tSSID : %s\n\tPassword : %s",
38 (const char *)wifi_sta_cfg->ssid,
39 (const char *)wifi_sta_cfg->password);
40 break;
41 }
42 case WIFI_PROV_CRED_FAIL:
43 {
44 wifi_prov_sta_fail_reason_t *reason = (wifi_prov_sta_fail_reason_t *)event_data;
45 ESP_LOGE(BT_TAG, "Provisioning failed!\n\tReason : %s"
46 "\n\tPlease reset to factory and retry provisioning",
47 (*reason == WIFI_PROV_STA_AUTH_ERROR) ? "Wi-Fi station authentication failed" : "Wi-Fi access-point not found");
48
49 prov_retry_num++;
50 if (prov_retry_num >= PROV_MGR_MAX_RETRY_CNT)
51 {
52 ESP_LOGI(BT_TAG, "Failed to connect with provisioned AP, reseting provisioned credentials and restarting...");
53 wifi_config_t wifi_cfg = {0};
54 esp_err_t err = esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg);
55 if (err != ESP_OK)
56 {
57 ESP_LOGE(BT_TAG, "Failed to set wifi config, 0x%x", err);
58 }
59
60 xEventGroupSetBits(s_wifi_event_group, RESTART);
61 }
62
63 break;
64 }
65 case WIFI_PROV_CRED_SUCCESS:
66 ESP_LOGI(BT_TAG, "Provisioning successful");
67 break;
68 case WIFI_PROV_END:
69 // De-initialize manager once provisioning is finished and restart
70 ESP_LOGI(BT_TAG, "Provisioning end");
71 wifi_prov_mgr_deinit();
72 xEventGroupSetBits(s_wifi_event_group, RESTART);
73 break;
74 default:
75 break;
76 }
77 }
78
79 ESP_LOGI(BT_TAG, "end bt_event_handler: -------------------");
80}

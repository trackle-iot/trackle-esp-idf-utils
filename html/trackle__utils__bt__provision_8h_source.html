31/* Event handler for catching system events */
32static void bt_event_handler(void *arg, esp_event_base_t event_base,
33 int32_t event_id, void *event_data)
34{
35 ESP_LOGI(BT_TAG, "----------------------------------------");
36 ESP_LOGI(BT_TAG, "bt event_handler: %s %d", event_base, event_id);
37 ESP_LOGI(BT_TAG, "----------------------------------------");
38
39 if (event_base == WIFI_PROV_EVENT)
40 {
41 switch (event_id)
42 {
43 case WIFI_PROV_START:
44 ESP_LOGI(BT_TAG, "Provisioning started");
45 xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
46 xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_RUN);
47 break;
48 case WIFI_PROV_CRED_RECV:
49 {
50 wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t *)event_data;
51 ESP_LOGI(BT_TAG, "Received Wi-Fi credentials"
52 "\n\tSSID : %s\n\tPassword : %s",
53 (const char *)wifi_sta_cfg->ssid,
54 (const char *)wifi_sta_cfg->password);
55 break;
56 }
57 case WIFI_PROV_CRED_FAIL:
58 {
59 wifi_prov_sta_fail_reason_t *reason = (wifi_prov_sta_fail_reason_t *)event_data;
60 ESP_LOGE(BT_TAG, "Provisioning failed!\n\tReason : %s"
61 "\n\tPlease reset to factory and retry provisioning",
62 (*reason == WIFI_PROV_STA_AUTH_ERROR) ? "Wi-Fi station authentication failed" : "Wi-Fi access-point not found");
63
64 prov_retry_num++;
65 if (prov_retry_num >= PROV_MGR_MAX_RETRY_CNT)
66 {
67 ESP_LOGI(BT_TAG, "Failed to connect with provisioned AP, reseting provisioned credentials and restarting...");
68 wifi_config_t wifi_cfg = {0};
69 esp_err_t err = esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg);
70 if (err != ESP_OK)
71 {
72 ESP_LOGE(BT_TAG, "Failed to set wifi config, 0x%x", err);
73 }
74 xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
75 xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_ERR);
76 xEventGroupSetBits(s_wifi_event_group, RESTART);
77 }
78
79 break;
80 }
81 case WIFI_PROV_CRED_SUCCESS:
82 ESP_LOGI(BT_TAG, "Provisioning successful");
83 xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
84 xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_OK);
85 break;
86 case WIFI_PROV_END:
87 // De-initialize manager once provisioning is finished and restart
88 ESP_LOGI(BT_TAG, "Provisioning end");
89 wifi_prov_mgr_deinit();
90 xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
91 xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_NO);
92 xEventGroupSetBits(s_wifi_event_group, RESTART);
93 break;
94 default:
95 break;
96 }
97 }
98
99 ESP_LOGI(BT_TAG, "end bt_event_handler: -------------------");
100}

110static int btPostCbClaimCode(const char *args)
111{
112 char *key = strtok(args, ",");
113 if (key == NULL || strcmp(key, "cc") != 0)
114 {
115 ESP_LOGE("cc", "Invalid key for setting claim code");
116 return -1;
117 }
118 char *claimCode = strtok(NULL, ",");
119 if (key == NULL || strlen(claimCode) != 63)
120 {
121 ESP_LOGE("cc", "Invalid claim code");
122 return -1;
123 }
124 ESP_LOGE("cc", "Claim code received successfully:");

126 trackleSetClaimCode(trackle_s, claimCode);

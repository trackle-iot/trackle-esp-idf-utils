37static void event_handler(void *arg, esp_event_base_t event_base,
38 int32_t event_id, void *event_data)
39{
40 if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_START)
41 {
42 ESP_LOGI(WIFI_TAG, "Wifi started.....");
43
44 wifi_mode_t currentMode;
45 esp_wifi_get_mode(&currentMode);
46 if (currentMode == WIFI_MODE_STA)
47 {
48 ESP_LOGI(WIFI_TAG, "Connecting to the AP");
49 xEventGroupSetBits(s_wifi_event_group, WIFI_TO_CONNECT_BIT); // connettiti
50 // esp_wifi_connect();
51 timeout_connect_wifi = millis();
52 }
53 else if (currentMode == WIFI_MODE_APSTA)
54 {
55 ESP_LOGI(WIFI_TAG, "APMode, not connecting....");
56 xEventGroupClearBits(s_wifi_event_group, WIFI_TO_CONNECT_BIT); // non cercare di riconnetterti
57 }
58 }
59 else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_DISCONNECTED)
60 {
61 wifi_event_sta_disconnected_t *event = (wifi_event_sta_disconnected_t *)event_data;
62 ESP_LOGW(WIFI_TAG, "Wifi disconnection event: %d...", event->reason);
63 // wifi_err_reason_t
64
65 // esp_wifi_connect();
66 timeout_connect_wifi = millis() + CHECK_WIFI_TIMEOUT;
67 xEventGroupClearBits(s_wifi_event_group, WIFI_CONNECTED_BIT);
68 }
69 else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP)
70 {
71 ip_event_got_ip_t *event = (ip_event_got_ip_t *)event_data;
72 ESP_LOGW(WIFI_TAG, "Got ip:" IPSTR, IP2STR(&event->ip_info.ip));
73 s_retry_num = 0;
74 xEventGroupSetBits(s_wifi_event_group, WIFI_CONNECTED_BIT);
75 }
76}

78void wifi_init()
79{
80 ESP_LOGI(WIFI_TAG, "wifi_init....");

82 // init event group

89 ESP_ERROR_CHECK(esp_event_handler_register(IP_EVENT, IP_EVENT_STA_GOT_IP, &event_handler, NULL));
90
91 sta_netif = esp_netif_create_default_wifi_sta();
92 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
93 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
94 esp_wifi_set_storage(WIFI_STORAGE_FLASH);
95}

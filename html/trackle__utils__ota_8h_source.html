64 vTaskDelay(1000 / portTICK_PERIOD_MS);

67 else

69 tracklePublishSecure(OTA_EVENT_NAME, "failed");
70 xEventGroupClearBits(s_wifi_event_group, OTA_UPDATING); //stop updating

91 }

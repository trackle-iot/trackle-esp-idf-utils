43  int32_t event_id, void *event_data)
44 {
45  ESP_LOGI(BT_TAG, "----------------------------------------");
46  ESP_LOGI(BT_TAG, "bt event_handler: %s %d", event_base, event_id);
47  ESP_LOGI(BT_TAG, "----------------------------------------");
48 
49  if (event_base == WIFI_PROV_EVENT)
50  {
51  switch (event_id)
52  {
53  case WIFI_PROV_START:
54  ESP_LOGI(BT_TAG, "Provisioning started");
55  xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
56  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_RUN);
57  break;
58  case WIFI_PROV_CRED_RECV:
59  {
60  wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t *)event_data;
61  ESP_LOGI(BT_TAG, "Received Wi-Fi credentials"
62  "\n\tSSID : %s\n\tPassword : %s",
63  (const char *)wifi_sta_cfg->ssid,
64  (const char *)wifi_sta_cfg->password);
65  break;
66  }
67  case WIFI_PROV_CRED_FAIL:
68  {
69  wifi_prov_sta_fail_reason_t *reason = (wifi_prov_sta_fail_reason_t *)event_data;
70  ESP_LOGE(BT_TAG, "Provisioning failed!\n\tReason : %s"
71  "\n\tPlease reset to factory and retry provisioning",
72  (*reason == WIFI_PROV_STA_AUTH_ERROR) ? "Wi-Fi station authentication failed" : "Wi-Fi access-point not found");
73 
74  prov_retry_num++;
75  if (prov_retry_num >= PROV_MGR_MAX_RETRY_CNT)
76  {
77  ESP_LOGI(BT_TAG, "Failed to connect with provisioned AP, reseting provisioned credentials and restarting...");
78  wifi_config_t wifi_cfg = {0};
79  esp_err_t err = esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg);
80  if (err != ESP_OK)
81  {
82  ESP_LOGE(BT_TAG, "Failed to set wifi config, 0x%x", err);
83  }
84  xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
85  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_ERR);
86  xEventGroupSetBits(s_wifi_event_group, RESTART);
87  }
88 
89  break;
90  }
91  case WIFI_PROV_CRED_SUCCESS:
92  ESP_LOGI(BT_TAG, "Provisioning successful");
93  xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
94  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_OK);
95  break;
96  case WIFI_PROV_END:
97  // De-initialize manager once provisioning is finished and restart
98  ESP_LOGI(BT_TAG, "Provisioning end");
99  wifi_prov_mgr_deinit();
100  xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN);
101  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_NO);
102  xEventGroupSetBits(s_wifi_event_group, RESTART);
103  break;
104  default:
105  break;
106  }
107  }

109  ESP_LOGI(BT_TAG, "end bt_event_handler: -------------------");
